m = ViT(
    image_size=100,
    patch_size=10,
    num_classes=2,
    dim=128,
    depth=6,
    heads=12,
    mlp_dim=64,

              nn.MSELoss())

# Training loop

model.fit(train_dataset, epoch=100, batch_size=32, transform=transform)

from create_data import *

def predict(img):
    process = cv.cvtColor(img, cv.COLOR_BGR2GRAY)
    process = process[np.newaxis, ...]
    process = transform(torch.tensor(process)).unsqueeze(0)
    seg = model.predict(process).cpu().detach().squeeze().int().numpy()
    return seg

    cv.circle(img, seg, 8, (0, 255, 255), 2)

    data.append(img)
